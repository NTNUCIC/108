print('myadd:')

def myadd(*args):
    s=0
    for i in args:
        s+=i
    return s

a=[int(i) for i in input().split()]
print(myadd(*a))

file

寫檔


f1=open('input5-1.txt','r')
f2=open('output5-1.txt','w')

for line in f1:
    print(line,end='')
    f2.write(line)

f1.close()#關檔很重要
f2.close()#關檔很重要

with

自動close()


with open('input5-1.txt','r') as f:
    for line in f:
        print(line,end='')
