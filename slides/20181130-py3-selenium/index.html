以get依序瀏覽


from selenium import webdriver
from time import sleep

urls = ['https://www.google.com',
        'https://www.ntnu.edu.tw/',
        'https://www.ntnucic.club/Courses/']

browser = webdriver.Chrome()
browser.maximize_window()

for url in urls:
    browser.get(url) 
    sleep(3)

browser.close()

close()
